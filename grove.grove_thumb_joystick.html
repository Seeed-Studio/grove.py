  • Grove - Thumb Joystick

Examples

import time
from grove.grove_thumb_joystick import GroveThumbJoystick

# connect to alalog pin 2(slot A2)
PIN = 2

sensor = GroveThumbJoystick(PIN)

while True:
    x, y = sensor.value
    if x > 900:
        print('Joystick Pressed')
    print("X, Y = {0} {1}".format(x, y))
    time.sleep(.2)
