def main():
    from grove.helper import SlotHelper
    sh = SlotHelper(SlotHelper.GPIO)
    pin = sh.argv2pin()

    led = GroveLed(pin)

    while True:
        led.on()
        time.sleep(1)
        led.off()
        time.sleep(1)
