Examples

from grove.grove_switch import GroveSwitch
import time

# connect to pin 5 (slot D5)
PIN = 5
swicth = GroveSwitch(PIN)

while True:
    if swicth.state:
        print("high")
    else:
        print("low")
    time.sleep(1)
