Examples

import time
from grove.grove_rotary_angle_sensor import GroveRotaryAngleSensor

# connect to alalog pin 2(slot A2)
PIN = 2

sensor = GroveRotaryAngleSensor(PIN)

while True:
    print('Rotary Value: {}'.format(sensor.value))
    time.sleep(.2)
