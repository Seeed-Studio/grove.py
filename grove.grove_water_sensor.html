Examples

import time
from grove.grove_water_sensor import GroveWaterSensor

# connect to alalog pin 2(slot A2)
PIN = 2

sensor = GroveWaterSensor(PIN)

print('Detecting ...')
while True:
value = sensor.value
    if sensor.value > 800:
        print("{}, Detected Water.".format(value))
    else:
        print("{}, Dry.".format(value))

    time.sleep(.1)
