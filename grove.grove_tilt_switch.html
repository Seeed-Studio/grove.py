Examples

from grove.grove_tilt_switch import GroveTiltSwitch
import time

# connect to pin 5 (slot D5)
PIN = 5

swicth = GroveTiltSwitch(pin)

def on_trigger():
    print('Triggered')
def on_release():
    print("Released.")

swicth.on_trigger = on_trigger
swicth.on_release = on_release

while True:
    time.sleep(1)
