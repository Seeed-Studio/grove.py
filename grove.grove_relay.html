Examples

import time
from grove.grove_relay import GroveRelay

# connect to pin 5(slot D5)
PIN   = 5
relay = GroveRelay(PIN)

while True:
    relay.on()
    time.sleep(1)
    relay.off()
    time.sleep(1)
